#include <stdio.h>

int main()
{
	float income, tax;

    printf("Please enter your income: ");
    scanf("%f", &income);

    if(income>=0 && income <=45000)
    {
        tax = 0.15 * income;
    }

    if( income > 45000 && income <= 90000)
    {
        tax = 6750 + 0.20 * (income - 45000);
    }

    if( income > 90000 && income <= 140000)
    {
        tax = 15750 + 0.26 * (income - 90000);
    }

    if( income > 140000 && income <= 200000)
    {
        tax = 28750 + 0.29 * (income - 140000);
    }

    if( income > 200000)
    {
        tax = 46150 + 0.33 * (income - 200000);
    }

    if(income < 0)
    {
        printf("\nYour income is too low to calculate.");
        getchar();
        return 0;
    }

    printf("\nYour tax is %.2f\n", tax);

	getchar();
	return 0;
}
