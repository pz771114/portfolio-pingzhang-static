/* Source code for Lab4_a.c */

#include <stdio.h>

int main()
{
	/* Code block A */

	int totalMin, hours, minutes;

	printf("Enter the total minutes: ");
	scanf("%d",&totalMin);

	hours = totalMin / 60;
	minutes = totalMin % 60;

	printf("\n%d minutes = %d hours & %d minutes\n", totalMin, hours, minutes);

	/* Code block B */

	int numberA, numberB, numberC, sum;
	float average;

	printf("Enter three numbers as following\n");

	printf("First Number: ");
	scanf("%d", &numberA);

	printf("\nSecond Number: ");
	scanf("%d", &numberB);

	printf("\nThird Number: ");
	scanf("%d", &numberC);

	sum = numberA + numberB + numberC;

	average = sum / 3.0;


	printf("\nYou entered the numbers: %d, %d, %d\n",numberA, numberB, numberC);
	printf("The sum of three numbers is %d", sum);
	printf("\nThe average of three numbers is %.4f\n", average);


	getchar();
	return 0;
}	


/* Source code for Lab4_b.c */

#include <stdio.h>
#include <math.h>

int main()
{
	float canOne_diameter, canOne_height;
	float canTwo_diameter, canTwo_height;
	float numberOfTripsOne, numberOfTripsTwo;

	const int TUBE_SIZE = 1000;
	const float PI = 3.1416;

	printf("Enter the diameter and height for Can One\n");
	printf("\nDiameter: ");
	scanf("%f", &canOne_diameter);
	printf("\nHeight: ");
	scanf("%f", &canOne_height);

	numberOfTripsOne = (TUBE_SIZE * 4 )/ (PI * pow(canOne_diameter, 2.0) * canOne_height);
	printf("Can one takes %d trips to fill the tub.", ceil(numberOfTripsOne));

	printf("\nEnter the diameter and height for Can Two");
	printf("\nDiameter: ");
	scanf("%f", &canTwo_diameter);
	printf("\nHeight: ");
	scanf("%f", &canTwo_height);
	
	numberOfTripsTwo = (TUBE_SIZE * 4 )/ (PI * pow(canTwo_diameter, 2.0) * canTwo_height);
	printf("\nCan two takes %d trips to fill the tub.", ceil(numberOfTripsOne));

	if(numberOfTripsOne > numberOfTripsTwo)
	{
		printf("\nCan one takes the least amount of trips.");
	}else
	{
		printf("\nCan two takes the least amount of trips.");
	}

	return 0;
}
